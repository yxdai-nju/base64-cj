 52 |     let pos = 0;
 53 |     const len = string.length;
 54 |     const out = [];
 55 |     let at = 0; // output position
 56 |     let tlen = Math.max(32, len + (len >> 1) + 7); // 1.5x size
 57 |     let target = new Uint8Array((tlen >> 3) << 3); // ... but at 8 byte offset
 58 |     while (pos < len) {
 59 |       let value = string.charCodeAt(pos++);
 60 |       if (value >= 0xd800 && value <= 0xdbff) {
 61 |         // high surrogate
 62 |         if (pos < len) {
 63 |           const extra = string.charCodeAt(pos);
 64 |           if ((extra & 0xfc00) === 0xdc00) {
 65 |             ++pos;
 66 |             value = ((value & 0x3ff) << 10) + (extra & 0x3ff) + 0x10000;
 67 |           }
 68 |         }
 69 |         if (value >= 0xd800 && value <= 0xdbff) {
 70 |           continue; // drop lone surrogate
 71 |         }
 72 |       }
 73 |       // expand the buffer if we couldn't write 4 bytes

 75 |         tlen += 8; // minimum extra
 76 |         tlen *= (1.0 + (pos / string.length) * 2); // take 2x the remaining
